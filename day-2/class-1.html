git --version - check the version of the git

git config --global user.name 'username' - add the user name

git config --global user.email 'ouremail' - add the email

git init - Initialise the repository/project

git add . - add all the changes to the repository

git add file-name - add the changes taht are made in that file to the repository

git commit -m 'added a button/paragraph/page'

git remote add origin (folder url path) - add remote repository as origin

git branch -M main - Create a main branch

git branch homepage - create a branch with name homepage

git merge aboutpage - merge the aboutpage branch with the current page branch

git checkout aboutpage - switch to the specific branch

git branch - shows all the branches in our origin/repository

git push -u origin main - pushes the code in the main branch

git push -u origin branch_name - push the code to the repository

git pull - pull/update the code in our device after merging

git fetch -pull/update the code without merging

git clone url - clone the project

git log - show the commit history

git log branch_name - show the commit history of that perticular branch

git log --all - shows commit history of all the branches

git log --oneline - show commit history in a condesced manner

git remote - show origin we have added or not

git remote -v - shows the remote repository

git remote remove origin - remove the remote repository
